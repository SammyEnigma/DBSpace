CREATE VIEW [dbo].[vNursingInfants]
AS
SELECT        P.PersonID, P.LastName, P.FirstName, P.Age, P.Gender,P.NursingInfant
FROM            dbo.Person AS P
WHERE        (P.NursingInfant = 1)
       AND [P].[isClient] = 1


GO
